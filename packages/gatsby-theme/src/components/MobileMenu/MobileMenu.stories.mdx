import { useState } from "react"
import { Meta, Story, ArgsTable, Canvas } from "@storybook/addon-docs/blocks"
import { Button } from "ui"
import MobileMenu from "."

export const mainMenu = [
  {
    name: "Home",
    to: "/",
  },
  {
    name: "About",
    to: "/",
  },
  {
    name: "JAMSTACK",
    to: "/",
  },
  {
    name: "CMS",
    to: "/",
  },
  {
    name: "WEB DEVELOPMENT",
    to: "/",
  },
  {
    name: "Contact",
    to: "/",
  },
]

export const BComponent = (args) => {
  const [show, handleShow] = useState(false)
  return (
    <>
      <Button onClick={() => handleShow(!show)}>Toggle</Button>
      <MobileMenu
        showSidebar={show}
        handleShowSidebar={() => handleShow(!show)}
        {...args}
      >
        <MobileMenu.Navbar>
          {mainMenu &&
            mainMenu.map((item) => (
              <MobileMenu.Item key={item.name} to={item.to}>
                {item.name}
              </MobileMenu.Item>
            ))}
        </MobileMenu.Navbar>
      </MobileMenu>
    </>
  )
}

<Meta
  title='components/MobileMenu'
  component={MobileMenu}
  subcomponents={{
    MobileMenuItem: MobileMenu.Item,
    MobileMenuNavbar: MobileMenu.Navbar,
  }}
/>
